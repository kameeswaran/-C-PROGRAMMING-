#include <stdio.h>

int main() {
    int n;
    int first = 0;
    intsecond = 1; 
    int next; 
    int i;

    printf("Enter the number of terms: ");
    scanf("%d", &n);

    printf("Fibonacci Series: ");

    for (i = 0; i < n; i++) {
        if (i <= 1) {
            next = i;
        } else {
            next = first + second;
            first = second;
            second = next;
        }
        printf("%d ", next);
    }

    return 0;
}

OUTPUT:
Enter the number of terms: 8

Fibonacci Series: 0 1 1 2 3 5 8 13
