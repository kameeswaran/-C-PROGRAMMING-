#include <stdio.h>

int main() {
    float radius, area;
    printf("Enter the radius of the circle: ");
    scanf("%f", &radius);
    area = 3.14 * radius * radius;
    printf("Area of Circle = %.2f\n", area);

    return 0;
}
 
OUTPUT:
Enter the radius of the circle: 7
Area of Circle = 153.86
